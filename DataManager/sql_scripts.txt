Добавление таблицы для сравнения моделей

CREATE TABLE predictions_all_models (
    id SERIAL PRIMARY KEY,
    match_id INTEGER NOT NULL,
    is_main_model BOOLEAN NOT NULL,
    model_name VARCHAR(255) NOT NULL,
    predicted_result matchresult NOT NULL
    
    -- Добавление внешнего ключа
    CONSTRAINT fk_predictions_all_models_match 
        FOREIGN KEY (match_id) 
        REFERENCES matches(match_id)
        ON DELETE CASCADE
);
